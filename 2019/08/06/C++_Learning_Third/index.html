冒泡排序算法

int i, j, exchange, sum[30];
for( i=0 ; i<30 ; i++){
for( j=i ; j < 30 ; j++){
if (sum[i]<sum[j]){
exchange=sum[i];
sum[i]=sum[i+1];
sum[i+1]=exchange;
}
}
}
